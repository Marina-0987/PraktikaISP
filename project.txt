-- Таблица "Должность"
CREATE TABLE Продажа товара (
    Код INT PRIMARY KEY,
    Наименование товара VARCHAR(255)
    Срок годности(при наличии) DATE,
    Цена INT,
    Категория товара VARCHAR(255),
    Дата продажи DATE, 
    FOREIGN KEY (Сотрудник) REFERENCES Сотрудники(Номер)
);

-- Таблица "Бракованные товары"
CREATE TABLE Бракованные товары (
    Номер INT PRIMARY KEY,
    Наименование товара VARCHAR(255),
    Срок годности(при наличии) DATE,
    Цена INT,
    Категория товара VARCHAR(255),
    Причина бракованности VARCHAR(255),
    Дата поступления на склад DATE,
    Количество бракованных товаров INT
);

-- Таблица "Сотрудники"
CREATE TABLE Сотрудники (
    Номер INT PRIMARY KEY,
    ФИО VARCHAR(255),
    Должность VARCHAR(255),
    Номер телефона INT
);

-- Таблица "Закупка товара"
CREATE TABLE Закупка товара (
    Номер INT PRIMARY KEY,
    Наименование товара VARCHAR(255),
    Срок годности(при наличии) DATE,
    Цена INT,
    Категория товара VARCHAR(255),
    Дата покупки DATE,
    FOREIGN KEY (Сотрудник) REFERENCES Сотрудники(Номер)
);

-- Таблица "Товар"
CREATE TABLE Товар (
    Номер INT PRIMARY KEY,
    Наименование товара VARCHAR(255),
    Срок годности(при наличии) DATE,
    Цена INT,
    Количество в наличии(на складе) INT,
    Дата поступления на склад DATE,
    Дата продажи DATE,
    Категория товара VARCHAR(255)
);

-- Таблица "Категория"
CREATE TABLE Категория (
    Номер INT PRIMARY KEY,
    Категория товара VARCHAR(255),
    Назначание VARCHAR(255)
);
